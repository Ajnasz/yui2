Calendar Widget

Default Single-Select Implementation

To allow for multiple selections on a 2-up Calendar, use the following code:

var cal1; function init() { cal1 = new YAHOO.widget.Calendar2up("cal1","cal1Container"); var customConfig = function() { this.Options.MULTI_SELECT = true; } cal1.setChildFunction("customConfig", customConfig); cal1.callChildFunction("customConfig"); cal1.render(); } <div id="cal1Container"></div>
